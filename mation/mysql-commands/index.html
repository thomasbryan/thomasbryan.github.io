Mysql Commands

  

export database:
$ mysqldump (database name) -u (user) -p > (sql dump file)
Enter password:

import database:
$ mysql (database name) -u (user) -p < (sql dump file)
Enter password:

connect to mysql:
$ mysql -h (host) -u (user) -p
Enter password:

create user:
mysql> create user '(name)'@'(host)' identified by '(password)';

set user permissions:
mysql> grant (permission) on (database name).(table) to '(user)'@'(host)';

activate permissions:
mysql> flush privileges;

find out information on table:
mysql> show table status like '%(table name)%';

find column name in tables:
mysql> SELECT TABLE_NAME, COLUMN_NAME FROM INFORMATION_SCHEMA.COLUMNS WHERE column_name LIKE '(field name)';

find out autoincrement value:
mysql> select Auto_increment from information_schema.tables where table_name = '(table name)' and table_schema = database();

change auto increment value:
mysql> alter table '(table name)' auto_increment='(value)';

change row value:
mysql> update '(table name)' set (field name)='(value)' where (field name)='(value)';

insert row value:
mysql> insert into '(table name)' (field1, field2, field3) values '(value1, value2, value3)';

delete row:
mysql> delete from '(table name)' where (field name) = '(value)';
